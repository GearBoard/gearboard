  console.log(`Server running at http://localhost:${info.port}`);

App — apps/api/src/app.ts

const app = new Hono();
app.use(cors({ origin: env.BETTER_AUTH_TRUSTED_ORIGIN, credentials: true }));
app.on(["POST", "GET"], "/api/auth/**", (c) => auth.handler(c.req.raw));
app.get("/", (c) => c.text("Hello World!"));
app.route("/", apiRoutes);
app.onError((err, c) => {
  const { statusCode, message } = resolveHttpError(err);
  return c.json({ success: false, message }, statusCode);
});

Routes — apps/api/src/routes.ts

export const apiRoutes = new Hono()
  .route("/api/posts", postRoute)
  .route("/api/uploads", uploadRoute)
  .route("/api/comments", commentRoute)
  .route("/api/users", userRoute);

export type AppType = typeof apiRoutes;
Why routes.ts is separate from app.ts

app.ts uses .on() (wildcard auth handler) and .onError(). These break TypeScript's ability to accumulate route schema types via typeof app. routes.ts is a clean Hono chain — only .route() calls — so typeof apiRoutes resolves to the full typed schema used by hc<AppType> on the frontend.

Config — env, auth, prisma

🔄 Request Lifecycle

Request → CORS middleware (global, all routes) → requireAuth (optional — applied per route) → requireAdmin (optional — role check) → zValidator("param" | "query" | "json", ZodSchema, validationHook) → handler: c.req.valid() → service() → c.json(successResponse(result), status) → app.onError() (catches thrown AppError / unknown errors)

Auth middleware — apps/api/src/common/middleware/auth.middleware.ts

export const requireAuth = createMiddleware<{ Variables: AppVariables }>(async (c, next) => {
  const session = await auth.api.getSession({ headers: c.req.raw.headers });
  if (!session?.user) return c.json({ success: false, message: "Unauthorized" }, 401);
  const dbUser = await userRepository.findById(session.user.id);
  if (!dbUser) return c.json({ success: false, message: "Unauthorized" }, 401);
  c.set("user", { id: dbUser.id, name: dbUser.name, image: dbUser.image ?? null,
    role: dbUser.role, email: dbUser.email, username: dbUser.username ?? null });
  await next();
});

Route example — apps/api/src/modules/post/post.route.ts

export const postRoute = new Hono<{ Variables: AppVariables }>()
  .get("/:id", zValidator("param", GetPostByIdParamsInputDTO, validationHook), async (c) => {
    const { id } = c.req.valid("param");
    const result = await getPostByIdService(id);
    return c.json(successResponse(result), 200);
  })
  .post("/", requireAuth, zValidator("json", CreatePostBodyInputDTO, validationHook), async (c) => {
    const user = c.get("user");
    const data = c.req.valid("json");
    const result = await createPostService(data, user.id);
    return c.json(successResponse(result, "Post created"), 201);
  });

Route methods must remain chained. Imperative postRoute.get(...) calls break TypeScript's schema inference.

📋 DTO & Validation

One DTO file per endpoint. Each file exports an Input DTO (Zod schema used in zValidator()) and an Output DTO (class with static toDTO()).

Input DTO — dto/create-post.dto.ts

export const CreatePostBodyInputDTO = z.object({
  title: z.string().trim().min(1, "Title is required").max(255, "Title too long"),
  description: z.string().trim().min(1, "Description is required"),
  tags: z.array(z.string().trim()).optional().default([])
    .transform((tags) => [...new Set(tags)]),    // deduplicate
  images: z.array(z.string().url("Invalid image URL")).optional().default([]),
});
export type CreatePostBody = z.infer<typeof CreatePostBodyInputDTO>;

Output DTO — dto/get-post-by-id.dto.ts

export class GetPostByIdOutputDTO {
  id!: string; title!: string; tags!: string[]; images!: string[];
  authorInfo!: { id: string; username: string | null; image: string | null };
  createdAt!: string; updatedAt!: string;

  static toDTO(post: Post): GetPostByIdOutputDTO {
    return {
      id: post.id, title: post.title,
      tags: post.tags.map((pt) => pt.tag.name),        // flatten nested relation
      images: post.images.map((pi) => pi.url),
      authorInfo: { id: post.user.id, username: post.user.username, image: post.user.image },
      createdAt: post.createdAt.toISOString(),
      updatedAt: post.updatedAt.toISOString(),
    };
  }
}

Validation hook — common/utils/validation-hook.ts

export function validationHook(result: { success: boolean; error?: ZodError }, c: Context) {
  if (!result.success) {
    const message = result.error?.issues[0]?.message ?? "Validation error";
    return c.json({ success: false, message }, 400);   // returns first error only
  }
}

🧩 Service Layer

One service file per endpoint. Services are thin: fetch from repo, authorise, return DTO. Business errors are thrown as typed AppError subclasses.

Read service

export async function createPostService(data: CreatePostBody, userId: string): Promise<CreatePostOutputDTO> {
  const post = await postRepository.create(data, userId);
  return CreatePostOutputDTO.toDTO(post);
}

Mutation service with ownership check

export async function updatePostService(id: string, data: UpdatePostBody, userId: string) {
  const post = await postRepository.findById(id);
  if (!post) throw new NotFoundError("Post not found");
  if (post.userId !== userId) throw new ForbiddenError("Forbidden");
  const updated = await postRepository.update(id, data);
  return UpdatePostOutputDTO.toDTO(updated);
}

Type definition — post.repository.ts

const postInclude = {
  user: { select: { id: true, username: true, image: true } },
  tags: { include: { tag: true } },
  images: true,

Soft delete pattern

// Every query filters deletedAt: null
prisma.post.findUnique({ where: { id, deletedAt: null }, include: postInclude });

// Delete sets the timestamp
async softDelete(id: string): Promise<void> {
  await prisma.post.update({ where: { id, deletedAt: null }, data: { deletedAt: new Date() } });
}

Pagination utility — common/utils/pagination.ts

export function getSkipTake(page = 1, limit = 10): { skip: number; take: number } {
  const take = Math.min(Math.max(1, limit), 100);  // clamp 1–100
  const skip = (Math.max(1, page) - 1) * take;
  return { skip, take };
}

🔌 API Client & Hono RPC

The frontend uses Hono RPC — no Axios, no codegen, no OpenAPI. The client is typed directly from AppType in the API source.

apps/web/src/shared/libs/api-client.ts

import { hc } from "hono/client";
import type { AppType } from "../../../../api/src/routes";

export const client = hc<AppType>(process.env.NEXT_PUBLIC_API_URL!);

export async function unwrap<T extends { data: unknown }>(
  resPromise: Promise<{ ok: boolean; json(): Promise<T> }>
): Promise<T["data"]> {
  const res = await resPromise;
  const body = await res.json();
  if (!res.ok)
    throw new Error((body as { message: string }).message ?? "Request failed");
  return body.data;
}

T["data"] (index type) resolves eagerly, giving concrete return types. A conditional type T extends ... ? D : never would defer resolution and lose type info.

How the RPC type inference chain works
  1. routes.ts builds a chained Hono instance — typeof apiRoutes accumulates all route schemas.
  2. AppType = typeof apiRoutes exported from routes.ts.
  3. Frontend imports AppType as a type-only import — no runtime code crosses the workspace boundary.
  4. hc<AppType>(url) produces a proxy object typed to match every endpoint, param, query, body, and response shape.
  5. client.api.posts[":id"].$get({ param: { id } }) is fully type-checked.

🪝 SWR Hooks

All data fetching goes through shared/hooks/. GETs use useSWR; mutations use useSWRMutation. Import from @/shared/hooks.

GET hook — hooks/posts.ts

export function useGetPostById(id: string) {
  return useSWR(["post", id], () =>
    unwrap(client.api.posts[":id"].$get({ param: { id } }))
  );
}
export function useGetPostList(query?: { page?: string; search?: string; tag?: string }) {
  return useSWR(["posts", query], () =>
    unwrap(client.api.posts.$get({ query: query ?? {} }))
  );
}

Mutation hook — hooks/posts.ts

export function useCreatePost() {
  return useSWRMutation(
    "posts",
    (_key: string, { arg }: { arg: { title: string; description: string; tags?: string[] } }) =>
      unwrap(client.api.posts.$post({ json: arg }))
  );
}

// Usage in a component:
const { trigger, isMutating } = useCreatePost();
await trigger({ title: "...", description: "..." });

📌 Add a Backend Endpoint

  1. DTO file — create apps/api/src/modules/<module>/dto/<action>-<resource>.dto.ts
    Export const <Action><Resource>InputDTO = z.object({...}) + inferred type.
    Export class <Action><Resource>OutputDTO with static toDTO(entity: EntityType).
    Add re-export to dto/index.ts.
  2. Service file — create service/<action>-<resource>.service.ts
    export async function <action><Resource>Service(...) — call repo, check ownership if needed, return DTO.
    Throw NotFoundError / ForbiddenError etc. for domain errors.
    Add re-export to service/index.ts.
  3. Route — add a chained method to <module>.route.ts: 
    .get("/new-path", requireAuth,
  zValidator("param", InputDTO, validationHook),
  async (c) => {
    const input = c.req.valid("param");
    const result = await newService(input);
    return c.json(successResponse(result), 200);
  }
)
    Must stay chained — separate route.get() calls break TypeScript type inference.
  4. No changes needed to routes.ts, app.ts, or AppType — typeof apiRoutes updates automatically.
  5. Frontend hook — add to apps/web/src/shared/hooks/<resource>.ts and re-export from hooks/index.ts.

📌 Add a Frontend Page

  1. Create route — apps/web/src/app/<path>/page.tsx
    Add "use client" directive if using hooks or state.
  2. Fetch data — import hooks from @/shared/hooks: 
    import { useGetPostList } from "@/shared/hooks";

const { data, isLoading } = useGetPostList({ page: "1", limit: "10" });
// data is typed: { data: GetPostByIdOutputDTO[], total, page, limit, totalPages }
  3. Feature components — if the page is complex, co-locate components in apps/web/src/features/<feature>/components/.
  4. Shared UI — use components from @/shared/components: Button, Input, Modal, Dropdown.
  5. Auth guard — use useSession() from @/shared/libs/auth-client and useRouter() to redirect unauthenticated users.
